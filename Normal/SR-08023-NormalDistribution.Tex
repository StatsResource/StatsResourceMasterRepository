\noindent \textbf{Compute the Z-Score}
\[ z_{o} = \frac{ x_{o} - \mu}{\sigma} \]
\end{framed}
\begin{itemize}
    \item $P(X\leq x_{o}) = P(Z \leq z_{o})$
    \item $P(Z \leq -z_{o} = P(Z \geq z_{o})$
\end{itemize}
