\subsection*{Probability}
\begin{itemize}


\item Exponential probability distribution:
\begin{equation*}
P(X \leq k) = \begin{cases}
1-e^{- k/\mu}, & k \ge 0, \\
0, & k < 0.
\end{cases}\qquad \left( \text{where  }
\mu = {1\over \lambda}\right)
\end{equation*}
\end{itemize}
