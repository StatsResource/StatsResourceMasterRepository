\begin{itemimize}

\noindent \textbf{Type 2} \\
\noindent Where parameters are 
\begin{description}
\item[shape] $\alpha > 0$
\item[rate] $\lambda > 0$
\end{description}
\[{\displaystyle MGF : \left(1-{\frac {t}{\lambda }}\right)^{-\alpha }{\text{ for }}t<\lambda } \]
